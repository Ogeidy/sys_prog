EX09:   PROC  OPTIONS  ( MAIN );
    DCL A   BIT  ( 5 )  INIT  ( ‘101’B );
    DCL B   DEC  FIXED;
    B = A;
END  EX09;  

Assembler:

EX09    START   0       #Начало программы#
        BALR    RBASE,0 Загрузить регистр базы             
        USING   *,RBASE Назначить регистр базой  


                        #Объявления переменных#


        DS      B
A       DC      B’00101’   
        DS      P                    
B       DC      P'0'     


RBASE   EQU     15
RRAB    EQU     5   

END    EX09             #Конец программы#
